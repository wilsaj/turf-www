/**
 * Takes a LinearRing and returns true if the coordinate pairs are listed clockwise.
 *
 * @module turf/isClockwise
 * @param {Array} ring an Array of coordinate pairs
 * @return {Boolean} `true` if coordinate pairs are listed clockwise, `false` if coordinate pairs are not listed clockwise
 * @example
 * var clockwiseRing = [
 *  [-149, 61],
 *  [-150, 61],
 *  [-150, 62],
 *  [-149, 62],
 *  [-149, 61]
 * ];
 * var counterClockwiseRing = [
 *  [-149, 61],
 *  [-149, 62],
 *  [-150, 62],
 *  [-150, 61],
 *  [-149, 61]
 * ];
 *
 * var isCW = isClockwise(clockwiseRing);
 * var counterCW = isClockwise(counterClockwiseRing);
 *
 * //=isCW
 * //=counterCW
 *
 */

module.exports = function(ring){
  var sum = 0;
  var i = 1;
  var len = ring.length;
  var prev,cur;
  while(i<len){
    prev = cur||ring[0];
    cur = ring[i];
    sum += ((cur[0]-prev[0])*(cur[1]+prev[1]));
    i++;
  }
  return sum > 0;
}
