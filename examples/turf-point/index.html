turf-point

Creates a geojson Point Feature based on an x and y coordinate. Properties can be added optionally.

<div id='map'></div>

<script>
L.mapbox.accessToken = 'pk.eyJ1IjoibW9yZ2FuaGVybG9ja2VyIiwiYSI6Ii1zLU4xOWMifQ.FubD68OEerk74AYCLduMZQ';

var point = turf.point(-97.514914, 35.463453, {
    "title": "OKC Thunder",
    "description": "100 W Reno Ave, Oklahoma City",
    "marker-color": "#6BC65F",
    "marker-size": "large",
    "marker-symbol": "basketball"
});
var map = L.mapbox.map('map', 'morganherlocker.kgidd73k')
    .setView([35.463453, -97.514914], 12)
    .featureLayer.setGeoJSON(point);
</script>
