turf-isClockwise

Takes a ring and return true or false whether or not the ring is clockwise or counter-clockwise.

<div id='map'></div>

<script>
L.mapbox.accessToken = 'pk.eyJ1IjoibW9yZ2FuaGVybG9ja2VyIiwiYSI6Ii1zLU4xOWMifQ.FubD68OEerk74AYCLduMZQ';

var map = L.mapbox.map('map', 'morganherlocker.kgidd73k')
    .setView([35.466453, -97.514914], 4);

var polys = [
    turf.polygon([[
        [ -96.328125, 31.05293398570514],[ -96.328125, 43.068887774169625],
        [ -80.5078125, 43.068887774169625],[ -80.5078125, 31.05293398570514],
        [ -96.328125, 31.05293398570514]]
    ], {
        "fill": "#6BC65F",
        "stroke": "#6BC65F",
        "stroke-width": 5
    }),
    turf.polygon([[
        [ -119.53125, 30.14512718337613],[ -102.3046875, 30.14512718337613],
        [ -102.3046875, 42.81152174509788],[ -119.53125, 42.81152174509788],
        [ -119.53125, 30.14512718337613]]
    ], {
        "fill": "#25561F",
        "stroke": "#25561F",
        "stroke-width": 5
    })
];

var poly_1_clockwise = turf.isClockwise(polys[0].geometry.coordinates[0]);
var poly_2_clockwise = turf.isClockwise(polys[1].geometry.coordinates[0]);

polys[0].properties.title = poly_1_clockwise ? 'This is Clockwise' : 'This is not Clockwise';
polys[1].properties.title = poly_2_clockwise ? 'This is Clockwise' : 'This is not Clockwise';

var poly_fc = turf.featurecollection(polys);

L.mapbox.featureLayer().setGeoJSON(poly_fc).addTo(map);
</script>
